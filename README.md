1. Factorial

        #include<iostream>
        #include<string>
        using namespace std;
        int main()
        {
            int n, f = 1;
            cout << "\nFACTORIAL\n\n";
            cout << "Student's name: ";
            string name;
            cin >> name;
            cout << "\nEnter a Number: ";
            cin >> n;

            while (cin.fail())
            {
                cout << "\nInvalid Input\n\n";
                cout << "Enter a Number again: ";
                cin.clear();
                cin.ignore(1000, '\n');
                cin >> n;
            }

            for (int x = n; x > 0; x--)
            {
                f = x * f;
            }
            cout << "\nFactorial of " << n << " is " << f << endl;

        }
